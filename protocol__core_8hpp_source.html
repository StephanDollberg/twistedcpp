48  void run_protocol() {
49  auto self = this_protocol().shared_from_this();
50  boost::asio::spawn(_socket->get_io_service(),
51  [this, self](boost::asio::yield_context yield) {
52  _yield = boost::in_place(yield);
53 
54  try {
55  _socket->do_handshake(*_yield);
56 
57  while (_socket->is_open()) {
58  auto bytes_read =
59  _socket->async_read_some(asio_buffer(), yield);
60  checked_on_message(buffer_begin(),
61  std::next(buffer_begin(), bytes_read));
62  }
63  }
64  catch (boost::system::system_error& connection_error) {
65  handle_network_error(connection_error);
66  }
67 
68  this_protocol().on_disconnect();
69  });
70  }

80  auto self = this_protocol().shared_from_this();
81  // we pass our existing yield_context and not the io_service such that
82  // both contexts run under the same strand
83  boost::asio::spawn(*_yield, [this, duration, callable, self](
84  boost::asio::yield_context yield) {
85  boost::asio::high_resolution_timer timer(_socket->get_io_service(),
86  duration);
87  timer.async_wait(yield);
88 
89  // we need to exchange the yield context so that that the user can
90  // use the send_message function in the right context
91  boost::optional<boost::asio::yield_context> old_yield(
92  boost::in_place(*_yield));
93  _yield = boost::in_place(yield);
94  callable();
95  _yield = boost::in_place(*old_yield);
96  });
